class Solution {
public:
    int vowelConsonantScore(string s) {
        int v=0;
        int c=0;
        for(char m:s)
            {
              if(m>='a'&&m<='z') 
              {
                  if(m=='a'||m=='e'||m=='i'||m=='o'||m=='u')
                      v++;
                  else
                      c++;
              }
                  
            }
        if(c>0)
        {
            int score=floor(v/c);
            return score;
        }
        else
            return 0;
    }
};
