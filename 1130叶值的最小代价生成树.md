class Solution {
public:
    int mctFromLeafValues(vector<int>& arr) {
        int n = arr.size();
        vector<vector<int>> dp(n, vector<int>(n, INT_MAX));
        vector<vector<int>> maxnum(n, vector<int>(n, 0));
        for (int i = 0; i < n; i++) {
            maxnum[i][i] = arr[i];
            for (int j = i + 1; j < n; j++) {
                maxnum[i][j] = max(maxnum[i][j - 1], arr[j]);
            }
        }
        for (int i = 0; i < n; i++)
            dp[i][i] = 0;

        for (int len = 2; len <= n; len++) {
            for (int i = 0; i + len - 1 < n; i++) {
                int j = i + len - 1;
                for (int k = i; k < j; k++) {
                    dp[i][j] =
                        min(dp[i][j], dp[i][k] + dp[k + 1][j] +
                                          maxnum[i][k] * maxnum[k + 1][j]);
                }
            }
        }
        return dp[0][n - 1];
    }
};
