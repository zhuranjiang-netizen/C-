class Solution {
public:
    vector<long long> mergeAdjacent(vector<int>& nums) {
       stack<long long>st;
        for(int num:nums)
            {
                long long current=num;
                while(!st.empty()&&st.top()==current)
                    {
                        current+=st.top();
                        st.pop();
                    }
                st.push(current);
            }
        vector<long long>result;
        while(!st.empty())
            {
                result.push_back(st.top());
                st.pop();
            }
        reverse(result.begin(),result.end());
        return result;
    }
};
