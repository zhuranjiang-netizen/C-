public:
    vector<int> findDisappearedNumbers(vector<int>& nums) {
        int n = nums.size();

            if(nums[num - 1] > 0) {
                nums[num - 1] = -nums[num - 1];
            }
        }
        
        vector<int> res;
        // æ‰¾æ²¡æœ‰å‡ºç°çš„æ•°å­—

            }
        }
        
        return res;
    }
};
