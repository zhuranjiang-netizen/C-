class Solution {
public:
    bool hasAlternatingBits(int n) {
        vector<int>nums;
        while(n!=0)
        {
            int m=n%2;
            n=n/2;
            nums.push_back(m);
        }
        int y=nums[0];
        for(int i=1;i<nums.size();i++)
        {
            if(nums[i]==y)
            return false;
            else
            y=nums[i];
        }
        return true;
    }
};
