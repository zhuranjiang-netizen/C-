class Solution {
public:
    vector<vector<vector<TreeNode*>>> memo; // memo[start][end] å­˜å‚¨ BST

    vector<TreeNode*> generate(int start, int end) {
        if(start > end) return {nullptr};
        if(!memo[start][end].empty()) return memo[start][end];

        vector<TreeNode*> res;
        for(int i = start; i <= end; i++) {
            auto leftTrees = generate(start, i - 1);
            auto rightTrees = generate(i + 1, end);

            for(auto l : leftTrees) {
                for(auto r : rightTrees) {
                    TreeNode* root = new TreeNode(i, l, r);
                    res.push_back(root);
                }
            }
        }
        memo[start][end] = res; // ä¿å­˜ç»“æœ
        return res;
    }

    vector<TreeNode*> generateTrees(int n) {
        if(n == 0) return {};
        memo.resize(n + 2, vector<vector<TreeNode*>>(n + 2)); // äºŒç»´ memo
        return generate(1, n);
    }
};
