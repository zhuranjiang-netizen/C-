class Solution {
public:
    int dominantIndices(vector<int>& nums) {
        int n=nums.size();
        int result=0;
        vector<int>sum(n,0);
        sum[n-1]=nums[n-1];
        for(int i=n-2;i>=0;i--)
            {
                sum[i]=sum[i+1]+nums[i];
            }
        for(int i=n-2;i>=0;i--)
            {
                int average=sum[i+1]/(n-i-1);
                if(nums[i]>average)
                    result++;
            }
        return result;
    }
};
