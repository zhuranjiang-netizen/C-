class Solution {
public:
    int numSteps(string s) {
        int count = 0;

        while (s != "1") {
            if (s.back() == '0') {
                // å¶æ•° â†’ é™¤2
                s.pop_back();
            } else {
                // å¥‡æ•° â†’ +1
                int i = s.size() - 1;
                while (i >= 0 && s[i] == '1') {
                    s[i] = '0';
                    i--;
                }

                if (i < 0) {
                    s = "1" + s;
                } else {
                    s[i] = '1';
                }
            }
            count++;
        }

        return count;
    }
};
