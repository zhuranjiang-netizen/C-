class Solution {
public:
    unordered_map<long long, int> mp;

    int getnum(long long n) {
        if (n == 1) return 0;

        if (mp.count(n)) return mp[n];

        if (n % 2 == 0) {
            mp[n] = getnum(n / 2) + 1;
        } else {
            mp[n] = min(getnum(n - 1), getnum(n + 1)) + 1;
        }

        return mp[n];
    }

    int integerReplacement(int n) {
        return getnum((long long)n);
    }
};
