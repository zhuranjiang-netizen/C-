class Solution {
public:

    int numFactoredBinaryTrees(vector<int>& arr) {
        const long long MOD=1e9+7;
        sort(arr.begin(),arr.end());
        unordered_map<int,long long>dp;
        for(int x:arr)
        {
            dp[x]=1;
            for(int a:arr)
            {
                if(a>=x)
                break;
                if(x%a==0)
                {
                    int b=x/a;
                    if(dp.count(b))
                    {
                        dp[x]=(dp[x]+dp[a]*dp[b])%MOD;
                    }
                }
            }
        }
        long long ans=0;
        for(auto& [k,v]:dp)
        {
        ans=(ans+v)%MOD;
        }
        return ans;
    }
};
