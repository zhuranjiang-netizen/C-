# **maximum-non-negative-product-in-a-matrix**

## **解题代码（C++）

```cp
class Solution {
public:
    int maxProductPath(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();

        vector<vector<long long>> dpmax(m+1, vector<long long>(n+1));
        vector<vector<long long>> dpmin(m+1, vector<long long>(n+1));

        dpmax[1][1] = grid[0][0];
        dpmin[1][1] = grid[0][0];

   
        for(int i=2;i<=m;i++){
            dpmax[i][1] = dpmax[i-1][1] * grid[i-1][0];
            dpmin[i][1] = dpmax[i][1];
        }

        for(int j=2;j<=n;j++){
            dpmax[1][j] = dpmax[1][j-1] * grid[0][j-1];
            dpmin[1][j] = dpmax[1][j];
        }


        for(int i=2;i<=m;i++){
            for(int j=2;j<=n;j++){
                long long val = grid[i-1][j-1];

                long long a = dpmax[i-1][j] * val;
                long long b = dpmax[i][j-1] * val;
                long long c = dpmin[i-1][j] * val;
                long long d = dpmin[i][j-1] * val;

                dpmax[i][j] = max(max(a, b), max(c, d));
                dpmin[i][j] = min(min(a, b), min(c, d));
            }
        }

        if(dpmax[m][n] < 0) return -1;
        return dpmax[m][n] % 1000000007;
    }
};
解题思路：
记录下最大值与最小值，注意初始化的时候要注意，因为只有一条走法，另外在求最大与最小值的时候要全部列出来，
不能按正负分类讨论。
