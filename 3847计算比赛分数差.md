class Solution {
public:
    int scoreDifference(vector<int>& nums) {
        int result=0;
        bool m=true;
        for(int i=0;i<nums.size();i++)
        {
            if(nums[i]%2==1)
            {
                m=!m;
            }
            if((i+1)%6==0)
            m=!m;
            if(m==true)
            result+=nums[i];
            else
            result-=nums[i];
        }
    return result;
    }

};
