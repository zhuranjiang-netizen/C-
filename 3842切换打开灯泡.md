class Solution {
public:
    vector<int> toggleLightBulbs(vector<int>& bulbs) {
        vector<int>result;
        unordered_set<int>hash;
        for(int i=0;i<bulbs.size();i++)
        {
            if(hash.find(bulbs[i])!=hash.end())
            hash.erase(bulbs[i]);
            else
            hash.insert(bulbs[i]);
        }
        for(int a:hash)
        {
            result.push_back(a);
        }
        sort(result.begin(),result.end());
        return result;
    }
};
