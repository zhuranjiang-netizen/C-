class Solution {
public:
    int numOfSubarrays(vector<int>& arr, int k, int threshold) {
        int n=arr.size();
        if(n<k)
        return 0;
        vector<int>sum(n+1,0);
        int count=0;
        for(int i=1;i<=n;i++)
        {
            sum[i]=sum[i-1]+arr[i-1];
        }
        int left=0;
        for(int right=k;right<=n;right++)
        {
            double average=(sum[right]-sum[left])/k+0.0;
            if(average>=threshold)
            count++;
            left++;
        }
        return count;

    }
};
