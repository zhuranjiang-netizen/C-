class Solution {
public:
    int longestBalanced(string s) {
          int n = s.size();
        int ans = 0;

        for (int i = 0; i < n; i++) {

            int freq[26] = {0};
            int distinct = 0;
            int maxFreq = 0;

            for (int r = i; r < n; r++) {

                int idx = s[r] - 'a';

                if (freq[idx] == 0)
                    distinct++;

                freq[idx]++;
                if (freq[idx] > maxFreq)
                    maxFreq = freq[idx];

                int len = r - i + 1;

                // åˆ¤æ–­æ˜¯å¦å¹³è¡¡
                if (len % distinct == 0) {
                    if (maxFreq == len / distinct) {
                        ans = max(ans, len);
                    }
                }
            }
        }

        return ans;
    }
};
