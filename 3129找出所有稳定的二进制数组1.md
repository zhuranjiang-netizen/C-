# **find-all-possible-stable-binary-arrays-i**

## **解题代码（C++）**

```cpp
class Solution {
public:
    int numberOfStableArrays(int zero, int one, int limit) {
        const int MOD = 1'000'000'007;
        vector<vector<vector<long long>>> dp(zero + 1, vector<vector<long long>>(one + 1, vector<long long>(2,0)));

        // base case: 只有一段长度为 1 的数字
        for(int k=1; k<=limit && k<=zero; ++k) dp[k][0][0] = 1;
        for(int k=1; k<=limit && k<=one;  ++k) dp[0][k][1] = 1;

        for(int i=0;i<=zero;i++){
            for(int j=0;j<=one;j++){
                // 放 0
                if(i>0){
                    for(int k=1;k<=limit && k<=i;k++){
                        dp[i][j][0] = (dp[i][j][0] + (j>0 ? dp[i-k][j][1] : 0)) % MOD;
                    }
                }
                // 放 1
                if(j>0){
                    for(int k=1;k<=limit && k<=j;k++){
                        dp[i][j][1] = (dp[i][j][1] + (i>0 ? dp[i][j-k][0] : 0)) % MOD;
                    }
                }
            }
        }
        return (dp[zero][one][0] + dp[zero][one][1]) % MOD;
    }
};
解题思路：
dp[i][j][]的含义是在用了i个0和j个1并且以0或1结尾的满足条件的数目，注意将题目的条件转化为相同的连续的数字的长度<=limit。
