class Solution {
public:
    int kthSmallest(vector<vector<int>>& matrix, int k) {
        int n = matrix.size();

        int left = matrix[0][0];
        int right = matrix[n - 1][n - 1];

        while (left < right) {
            int mid = left + (right - left) / 2;

            if (countLessEqual(matrix, mid) >= k)
                right = mid;// mid æœ‰å¯èƒ½æ˜¯ç­”æ¡ˆ â†’ æ”¶ç¼©å³è¾¹ç•Œ
            else
                left = mid + 1;// mid ä¸è¡Œ â†’ æ’é™¤ mid
        }
        return left;
    }

    int countLessEqual(vector<vector<int>>& matrix, int mid) {
        int n = matrix.size();
        int i = n - 1, j = 0;
        int cnt = 0;

        while (i >= 0 && j < n) {
            if (matrix[i][j] <= mid) {
                cnt += i + 1;
                j++;
            } else {
                i--;
            }
        }
        return cnt;
    }
};
