class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int n = cardPoints.size();
        int m = n - k;

        vector<int> sum(n, 0);
        sum[0] = cardPoints[0];
        for (int i = 1; i < n; i++) {
            sum[i] = sum[i - 1] + cardPoints[i];
        }

        // å…¨éƒ¨æ‹¿èµ°
        if (m == 0) return sum[n - 1];

        int minsum = sum[m - 1];

        for (int i = m; i < n; i++) {
            minsum = min(minsum, sum[i] - sum[i - m]);
        }

        return sum[n - 1] - minsum;
    }
};
