class Solution {
public:
    int countOne(int n)
    {
        int count = 0;
        while (n != 0)
        {
            if (n % 2 == 1)
                count++;
            n = n / 2;
        }
        return count;
    }

    vector<int> sortByBits(vector<int>& arr) {

        sort(arr.begin(), arr.end(), [&](int a, int b) {

            int countA = countOne(a);
            int countB = countOne(b);

            if (countA == countB)
                return a < b;

            return countA < countB;
        });

        return arr;
    }
};
