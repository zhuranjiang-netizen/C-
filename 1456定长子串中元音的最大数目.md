class Solution {
public:
    int maxVowels(string s, int k) {
        int n=s.size();
        int result=0;
        if(n<=k)
        {
            for(int i=0;i<n;i++)
            {
              if(s[i]=='a'||s[i]=='e'||s[i]=='i'||s[i]=='o'||s[i]=='u')
              {
                result++;
              }
            }
            return result;
        }
        for(int i=0;i<k;i++)
        {
             if(s[i]=='a'||s[i]=='e'||s[i]=='i'||s[i]=='o'||s[i]=='u')
              {
                result++;
              }
        }
        int length=result;
        for(int i=k;i<n;i++)
        {
          int j=i-k;
          if(s[i]=='a'||s[i]=='e'||s[i]=='i'||s[i]=='o'||s[i]=='u')
          length++;
          if(s[j]=='a'||s[j]=='e'||s[j]=='i'||s[j]=='o'||s[j]=='u')
          length--;
          result=max(result,length);
        }
        return result;
    }
};
