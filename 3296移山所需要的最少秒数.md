## **解题代码（C++）**

```cpp
class Solution {
public:
    long long mountainHeight;
    vector<int> workerTimes;

    bool check(long long T)
    {
        long long total = 0;

        for(long long t : workerTimes)
        {
            long long x = 2 * T / t;
            long long k = (sqrt(1 + 4 * x) - 1) / 2;
            total += k;

            if(total >= mountainHeight)
                return true;
        }

        return false;
    }

    long long minNumberOfSeconds(int mountainHeight, vector<int>& workerTimes) 
    {
        this->mountainHeight = mountainHeight;
        this->workerTimes = workerTimes;

        long long l = 0, r = 1e18;

        while(l < r)
        {
            long long mid = (l + r) / 2;

            if(check(mid))
                r = mid;
            else
                l = mid + 1;
        }

        return l;
    }
};
解题思路：
可以根据题目条件通过方程接触一个工人在T时间内能挖掘的最大高度，使用二分法去寻找满足条件的最时间。
