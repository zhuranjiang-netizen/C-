class Solution {
public:
    int minimumPrefixLength(vector<int>& nums) {
        int n=nums.size();
        if(n==1)
        return 0;
        int length=1;
        int current=nums[n-1];
        for(int i=n-2;i>=0;i--)
        {
          if(nums[i]>=current)
          {
            break;
          }
          else
         {
          current=nums[i];
          length++;
         }
        }
        return n-length;
    }
};
