class Solution {
public:
    vector<int> getSumAbsoluteDifferences(vector<int>& nums) {
        int n = nums.size();
        vector<int> result;
        vector<int> sum(n, 0);
        sum[0] = nums[0];
        for (int i = 1; i < n; i++) {
            sum[i] = nums[i] + sum[i - 1];
        }
        int index = 0;
        for (int i = 0; i < n; i++) {
            if (i == 0)
                result.push_back(sum[n - 1] - sum[0] - nums[0] * (n - 1));
            else
                result.push_back(i * nums[i] - sum[i - 1] + sum[n - 1] -
                                 sum[i] - (n - i - 1) * nums[i]);
        }
        return result;
    }
};
