class Solution {
public:
    int minimumTotal(vector<vector<int>>& triangle) {
        int n = triangle.size();
        vector<vector<int>> dp(n);
        for(int i = 0; i < n; i++)
            dp[i].resize(i+1, INT_MAX);

        dp[0][0] = triangle[0][0];

        for(int i = 1; i < n; i++) {
            for(int j = 0; j <= i; j++) {
                if(j == 0) dp[i][0] = dp[i-1][0] + triangle[i][0];
                else if(j == i) dp[i][i] = dp[i-1][i-1] + triangle[i][i];
                else dp[i][j] = min(dp[i-1][j-1], dp[i-1][j]) + triangle[i][j];
            }
        }

        int minpath = INT_MAX;
        for(int i = 0; i < n; i++)
            minpath = min(minpath, dp[n-1][i]);

        return minpath;
    }
};
