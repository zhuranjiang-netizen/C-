class Solution {
public:
    double champagneTower(int poured, int query_row, int query_glass) {
        double dp[101][101]={0};
        dp[0][0]=poured;
        for(int i=0;i<query_row;i++)
        {
        for(int j=0;j<=i;j++)
        {
         if(dp[i][j]>1.0)
         {
            double overflow=(dp[i][j]-1.0)/2;
            dp[i+1][j]+=overflow;
            dp[i+1][j+1]+=overflow;
            dp[i][j]=1.0;
         }   
        }
        }
        return dp[query_row][query_glass]>1.0?1.0:dp[query_row][query_glass];
    }
};
