class Solution {
public:
    int sortelement(vector<int>& nums, int left, int right) {
        // 1️⃣ 递归出口
        if (left == right) {
            return nums[left];
        }

        int mid = (left + right) / 2;

        // 2️⃣ 分
        int leftnum  = sortelement(nums, left, mid);
        int rightnum = sortelement(nums, mid + 1, right);

        // 3️⃣ 治
        if (leftnum == rightnum) {
            return leftnum;
        }

        // 4️⃣ 合并：在整个区间统计
        int countLeft = 0, countRight = 0;
        for (int i = left; i <= right; i++) {
            if (nums[i] == leftnum)  countLeft++;
            if (nums[i] == rightnum) countRight++;
        }

        return countLeft > countRight ? leftnum : rightnum;
    }

    int majorityElement(vector<int>& nums) {
        return sortelement(nums, 0, nums.size() - 1);
    }
};**
