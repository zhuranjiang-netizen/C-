class Solution {
public:
    int maxProduct(vector<int>& nums) {
     int n=nums.size();
     vector<int>dpmax(n,0);
     vector<int>dpmin(n,0);
     dpmax[0]=nums[0];
     dpmin[0]=nums[0];
     for(int i=1;i<n;i++)
     {
        dpmax[i]=max(max(dpmin[i-1]*nums[i],dpmax[i-1]*nums[i]),nums[i]);
        dpmin[i]=min(min(dpmin[i-1]*nums[i],dpmax[i-1]*nums[i]),nums[i]);
     }
     int result=INT32_MIN;
     for(int i=0;i<n;i++)
     {
      result=max(result,dpmax[i]);
     }
     return result;
    }
};
